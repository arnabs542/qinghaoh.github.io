public int[] finalPrices(int[] prices) {
    int[] result = Arrays.copyOf(prices, prices.length);
    Stack<Integer> st = new Stack<>();
    for (int i = 0; i < result.length; i++) {
        while (!st.isEmpty() && result[st.peek()] >= result[i]) {
            result[st.pop()] -= result[i];
        }
        st.push(i);
    }
    return result;
}
